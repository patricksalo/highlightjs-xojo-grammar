Public Function FriendlyNameForInterface(interfaceName As String) As String
  If interfaceName.Trim = "" Then
    Return ""
  End If

  // Normalize platform-specific network interface names.
  #If TargetMacOS Then
    Select Case interfaceName
    Case "en0"
      Return "Wi-Fi"
    Case "bridge100"
      Return "Thunderbolt Bridge"
    Else
      Return interfaceName
    End Select
  #ElseIf TargetWindows Then
    Return interfaceName.ReplaceAll("Ethernet adapter ", "")
  #Else
    Return interfaceName
  #EndIf
End Function

Class, properties, constants, arrays, and loops

Public Class PaymentSummary
  Public Const kCurrencyCode As String = "USD"

  Public Property JobID As String
  Public Property CustomerName As String
  Public Property SubtotalAmount As Double
  Public Property TaxAmount As Double
  Public Property AmountPaid As Double

  Public Function BalanceDue() As Double
    Return (SubtotalAmount + TaxAmount) - AmountPaid
  End Function

  Public Shared Function FromDictionary(values As Dictionary) As PaymentSummary
    Dim summary As New PaymentSummary

    summary.JobID = values.Lookup("JobID", "").StringValue
    summary.CustomerName = values.Lookup("CustomerName", "").StringValue
    summary.SubtotalAmount = values.Lookup("SubtotalAmount", 0.0).DoubleValue
    summary.TaxAmount = values.Lookup("TaxAmount", 0.0).DoubleValue
    summary.AmountPaid = values.Lookup("AmountPaid", 0.0).DoubleValue

    Return summary
  End Function
End Class

Dim payments() As PaymentSummary
