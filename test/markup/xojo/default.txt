Public Function FriendlyNameForInterface(interfaceName As String) As String
  If interfaceName.Trim = "" Then
    Return ""
  End If

  // Normalize platform-specific network interface names.
  #If TargetMacOS Then
    Select Case interfaceName
    Case "en0"
      Return "Wi-Fi"
    Case "bridge100"
      Return "Thunderbolt Bridge"
    Else
      Return interfaceName
    End Select
  #ElseIf TargetWindows Then
    Return interfaceName.ReplaceAll("Ethernet adapter ", "")
  #Else
    Return interfaceName
  #EndIf
End Function
